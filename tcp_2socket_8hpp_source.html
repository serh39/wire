439  Buffer& socket::read_until(uint8_t delimiter, Buffer& buf, std::error_code& ec, size_t max_size) noexcept {
440  uint8_t byte;
441  memory_view view{&byte, 1};
442  buf.clear();
443  while (read(1, view, ec), !ec) {
444  if (byte == delimiter) break;
445  if (max_size != 0 && buf.size() == max_size) break;
446  // Reinterpret cast is only way to safetly convert bytes between char and unsigned char representation.
447  buf.push_back(*reinterpret_cast<typename Buffer::value_type*>(&byte));
448  }
449  return buf;
450  }
